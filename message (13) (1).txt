  const WRAP_CLASS='tcl-logo-wrap';

  function makeWrap(){
    const p=Math.max(1,Math.floor(CFG.CELL_SIZE_PX));
    const wrap=document.createElement('span');
    wrap.className=WRAP_CLASS;
    Object.assign(wrap.style,{
      display:'inline-block',
      lineHeight:'0',
      transform:`translate(${CFG.SHIFT.x*p}px, ${CFG.SHIFT.y*p}px)`,
      padding:`${CFG.PAD.top*p}px ${CFG.PAD.right*p}px ${CFG.PAD.bottom*p}px ${CFG.PAD.left*p}px`
    });
    const el=document.createElement('tc-text');
    el.setAttribute('text', CFG.TEXT);
    el.setAttribute('size', String(p));
    el.setAttribute('gap', String(Math.max(0, CFG.LETTER_GAP)));
    el.setAttribute('color', CFG.COLOR);
    wrap.appendChild(el);
    return wrap;
  }

  function nukeOldChildren(host){
    // remove everything except our wrap
    host.querySelectorAll(`:scope > *:not(.${WRAP_CLASS})`).forEach(n=>n.remove());
    // kill any background logos
    host.style.background='none';
    host.style.backgroundImage='none';
  }

  function installGuard(host){
    if (host.__tclGuard) return;
    host.__tclGuard = new MutationObserver(()=>{
      // remove any re-inserted logos (including the specific SVG)
      host.querySelectorAll(':scope > img, :scope > picture, :scope > svg, :scope > canvas')
          .forEach(n=>{
            if (!n.closest(`.${WRAP_CLASS}`)) n.remove();
          });
      // generic: keep only our wrap
      host.querySelectorAll(`:scope > *:not(.${WRAP_CLASS})`).forEach(n=>n.remove());
      if (!host.querySelector(`:scope > .${WRAP_CLASS}`) && window.TeamClient){
        host.appendChild(makeWrap());
      }
    });
    host.__tclGuard.observe(host,{childList:true,subtree:false});
  }

  function replace(host){
    nukeOldChildren(host);
    if (!host.querySelector(`:scope > .${WRAP_CLASS}`)) {
      host.appendChild(makeWrap());
    }
    installGuard(host);
  }

  function apply(){
    if(!window.TeamClient) return false;
    const sels=Array.isArray(CFG.HOST_SEL)?CFG.HOST_SEL:[CFG.HOST_SEL];
    const nodes=sels.flatMap(s=>Array.from(document.querySelectorAll(s)));
    if(!nodes.length) return false;
    nodes.forEach(replace);
    return true;
  }

  let tries=0;(function loop(){
    if(apply()||++tries>CFG.MAX_TRIES) return;
    setTimeout(loop,200);
  })();
}

    function extractRoundTime(obj){
      let best=null;
      const Q=[{p:'',v:obj,d:0}], SEEN=new Set();
      const nameMatchLeft=/^(timeleft|time_left|remaining|left|countdown|timer)$/i;
      const nameMatchElapsed=/^(elapsed|played|survived|time_elapsed|round_time|game_time_played)$/i;
      const nameSoftContainsTime=/time/i;

      while(Q.length){
        const {p,v,d}=Q.shift();
        if(v==null||typeof v!=='object'||SEEN.has(v)||d>5) continue;
        SEEN.add(v);
        for(const k in v){
          const val=v[k], path=p?`${p}.${k}`:k;

          if(typeof val==='number' && isFinite(val)){
            let ms=null;
            if(val>=0 && val<=100000){ const msC=val*1000; if(msC<=MAX_REASONABLE_MS) ms=msC; }

              }
            }
          }
          if(val && typeof val==='object') Q.push({p:path,v:val,d:d+1});
        }
      }
      return best;
    }

      if(!hit) return false;
      if(hit.type==='left') setPerfStartFromLeft(hit.ms);

          else if(Array.isArray(msg) && msg[0]==='welcome' && msg[1]) pack=msg[1];

            if(sid!==lastSystemId){ lastSystemId=sid; lastSec=null; perfStart=null; }

            tryAdoptSocket(ws, pack);
            return;
          }

          if(isSurvival && (!activeWS || activeWS===ws)){
            tryAdoptSocket(ws, msg);
          }
        });

        return ws;
      };
      window.WebSocket.prototype=OG.prototype;
      ['OPEN','CLOSED','CONNECTING','CLOSING'].forEach(k=>window.WebSocket[k]=OG[k]);
    }

    // ---------- boot ----------
    buildUI();
    refreshTargets();
    applyZoomComp();
    requestAnimationFrame(tick);
  }catch(e){}

    const colHas=(c)=>{ for(let y=0;y<HEIGHT;y++) if(((rows[y]>>c)&1)===1) return true; return false; };
    let left=0; while(left<WIDTH && !colHas(left)) left++;
    let right=WIDTH-1; while(right>=0 && !colHas(right)) right--;
    return {left,right};
  }

  function makeSVG(text, opts={}){
    const px=Math.max(1, opts.size ?? DEFAULTS.size);
    const gap=Math.max(0, opts.gap ?? DEFAULTS.gap);
    const color=opts.color ?? DEFAULTS.color;

    let widthCells=0, info=[];
    for(const ch of text){
      if(ch===' '){ info.push({space:true,adv:3}); widthCells+=3+gap; continue; }
      const code=ch.codePointAt(0), {left,right}=metrics(code);
      const adv=(right<left?0:(right-left+1));
      info.push({code,left,right,adv}); widthCells+=adv+gap;
    }
    if(widthCells>0) widthCells-=gap;

    const W=widthCells*px, H=HEIGHT*px;
    let cursor=0, d='';
    for(const it of info){
      if(it.space){ cursor+=(it.adv+gap)*px; continue; }
      if(!it.adv){ cursor+=gap*px; continue; }
      const rows=GLYPHS[it.code];
      for(let y=0;y<HEIGHT;y++){
        const row=rows[y]>>>0;
        for(let x=it.left;x<=it.right;x++){
          if(((row>>x)&1)===1){ // LSB = leftmost (fixes mirroring)
            const X=cursor+(x-it.left)*px, Y=y*px;
            d+=`M${X} ${Y}h${px}v${px}h${-px}z`;
          }
        }
      }
      cursor+=it.adv*px + gap*px;
    }

    const svg=document.createElementNS('http://www.w3.org/2000/svg','svg');
    svg.setAttribute('width',W); svg.setAttribute('height',H);
    svg.setAttribute('viewBox',`0 0 ${W} ${H}`);
    const p=document.createElementNS('http://www.w3.org/2000/svg','path');
    p.setAttribute('fill',color); p.setAttribute('d',d||'');
    svg.appendChild(p);
    return svg;
  }

  window.TeamClient = {
    render(text,opts){ return makeSVG(text,opts); },
    renderInto(el,text,opts){
      const svg=makeSVG(text ?? (el.getAttribute('data-tc-text')||el.textContent||''), opts);
      el.innerHTML=''; el.appendChild(svg); el.classList.add('tc-font');
    }
  };

  if(!customElements.get('tc-text')){
    class TcText extends HTMLElement{
      static get observedAttributes(){ return ['text','size','gap','color']; }
      connectedCallback(){ this._render(); }
      attributeChangedCallback(){ this._render(); }
      _opts(){ return {
        size: this.hasAttribute('size') ? +this.getAttribute('size') : undefined,
        gap:  this.hasAttribute('gap')  ? +this.getAttribute('gap')  : undefined,
        color:this.getAttribute('color') || undefined
      }; }
      _render(){
        const txt=this.getAttribute('text')||this.textContent||'';
        this.textContent=''; this.appendChild(makeSVG(txt,this._opts()));
        this.classList.add('tc-font');
      }
    }
    customElements.define('tc-text', TcText);
  }

  const css=document.createElement('style');
  css.textContent=`.tc-font{display:inline-block;vertical-align:middle;line-height:0}
.tc-font svg{image-rendering:pixelated}`;
  document.head.appendChild(css);
}
function emoteCapacity(targetCap){
  try{
    // Resolve desired capacity (1..5), prefer localStorage
    let c;
    try{
      const ls = localStorage.getItem('sbcs_tc-emoteCapacity');
      c = Number(JSON.parse(ls ?? '5'));
    }catch{ c = Number(targetCap) || 5; }
    c = Math.max(1, Math.min(5, c || 5));

    let attempts = 0;
    (function tryPatch(){
      const CP = window.ChatPanel;
      if (!CP || !CP.prototype || typeof CP.prototype.typed !== 'function'){
        if (attempts++ < 300) return setTimeout(tryPatch, 100);
        return; // give up quietly
      }
      const P = CP.prototype;

      // Avoid double-patching
      if (P.__emoteCapPatched) return;

      // Provide API the rewritten function will call
      P.getEmotesCapacity = function(){ return c; };

      // Rewrite typed(): replace any small integer literal comparison with our getter
      const src = P.typed.toString();
      let out = src;

      // 1) Specific common case: ">= 4"
      out = out.replace(/>=\s*4\b/g, '>= this.getEmotesCapacity()');

      // 2) Broader: swap any small integer guard (1..10) after >=, >, <=, < nearby
      if (out === src){
        out = out.replace(/([<>]=?)\s*(\d+)\b/g, (m, op, num) => {
          const n = parseInt(num, 10);
          return (n >= 1 && n <= 10) ? `${op} this.getEmotesCapacity()` : m;
        });
      }

      // 3) If still unchanged, try a second pass limited to >= only (most likely)
      if (out === src){
        out = out.replace(/(>=)\s*(\d+)\b/g, (m, op, num) => {
          const n = parseInt(num, 10);
          return (n >= 1 && n <= 10) ? `${op} this.getEmotesCapacity()` : m;
        });
      }

      // Compile & swap only if we actually changed something
      if (out !== src){
        try{
          // Build a new function object from source text
          P.typed = Function('return ' + out)();
          P.__emoteCapPatched = true;
        }catch(e){
          console.error('[emoteCapacity] Failed to recompile ChatPanel.prototype.typed', e);
        }
      }else{
        // If we couldn’t find a literal to replace, still expose the capacity API
        // (kept for compatibility in case future code consults it)
        P.__emoteCapPatched = true;
      }
    })();
  }catch(e){
    console.error('[emoteCapacity] fatal', e);
  }
}
